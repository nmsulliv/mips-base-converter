.globl main
.data

	prompt: .asciiz "Enter a string of exactly 10 characters: "
	message: .asciiz "\nInput: "
	messagec: .asciiz "\nChar: "
	str: .space 32 # saves space for 32 bytes

.text

main:
	li $v0, 4 				#prompts the instructions to the user 
	la $a0, prompt
	syscall

	li $v0, 8				#get the number of characters to read from input 
	la $a0, str
	li $a1, 11 			
	syscall 
	
	li $v0, 4				#print out message
	la $a0, message
	syscall
	
	li $v0, 4				#print out input
	la $a0, str
	syscall
	
	li $v0, 4				#print out char message
	la $a0, messagec
	syscall
	
	li $v0, 11				#print out character of the string
	la $a0, str
	lb $a0, 0($a0)
	syscall
	
	addi $t0, $zero, 0		#initialize variables for while loop
	addi $t1, $zero, 10
	
	while:					#loop for each char of the string
		beq $t0, $t1, EXIT
		addi $t0, $t0, 1
		
		j while
	
	EXIT:
		li $v0, 10 			# exit program
		syscall
